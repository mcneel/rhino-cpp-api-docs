243  /*
244  Description:
245  Adds a Hook's law spring force bewtween particles that is proportional
246  to (distance - rest_length).
247  Parameters:
248  particle1 - [in]
249  particle2 - [in]
250  rest_length - [in]
251  rest length of the spring.
252  spring_constant - [in]
253  Spring force magnitude = spring_constant*(distance - rest_length).
254  When spring_constant > 0 and distance > rest_length, the force
255  attracts.
256  damping_constant - [in]
257  Damping force magnitude = damping_constant*|(delta velocity) o D|,
258  where D is a unit vector points from one particle to the other.
259  When damping_constant > 0 and the particles are moving towards
260  each other, the damping force repels.
261  When damping_constant > 0 and the particles are moving away from
262  each other, the damping force attracts.
263  */

305  /*
306  Description:
307  A unary force acts on every particle in the system.
308 
309  Parameters:
310  context - [in]
311  First parameter passed to force_function.
312 
313  force_function - [in]
314  A function that calculates the force on the particle.
315  If force_function() returns false, the solver will
316  stop.
317 
318  Example:
319  If you had 8, or perhaps 9, particles representing the planets in our
320  solar system, you were comfortable neglecting the effects the planets'
321  mass have on the sun's position and you chose a coordinate system
322  with the sun at the origin, then you could add a solar gravity force
323  doing something like the following.
324 
325  bool GetSolarGravity(
326  void*, // no context required
327  const class ON_ParticleSystem::ON_Particle* planet,
328  ON_3dVector* SolarGravityForce
329  )
330  {
331  // masses in kilograms and distances in meters
332  const double solar_mass = 1.9891e30; // kilograms
333  const double g = 6.67384e-11; // meters^3/(kilogram seconds^2)
334  ON_3dVector V = ON_3dPoint::Origin - planet->m_position;
335  double r = V.Length();
336  if ( !(r > 6.955e8) )
337  {
338  // Planet crashed into the sun!
339  // The sun's radius is 6.955e8 meters. This calculation
340  // ignores planets melting as they get too close to the sun.
341  return false;
342  }
343  *SolarGravityForce = (g*planet->m_mass*solar_mass/(r*r*r))*V; // r^3 because input V is not unitized
344  return true;
345  }
346 
347  ....
348  AddUnaryForce(0,GetSolarGravity);
349 
350  Remarks:
351  To add a global "gravity", it is more efficent to use SetGlobalGravityForce().
352  To add a global "drag" force, it is more efficient to use SetGlobalDragForce().
353  */
354  class ON_ParticleSystem::ON_Force* AddUnaryForce(

359  /*
360  Description:
361  A N-ary force acts on a fixed set of particles.
362 
363  Parameters:
364  force_function - [in]
365  A function that calculates the force on the particles
366  listed in the particles[] array parameter.
367  If force_function() returns false, the solver will
368  stop.
369 
370  context - [in]
371  First parameter passed to force_function.
372  If the information in your context consists of up to 4 doubles,
373  then you may use the ON_Force.m_context member as shown in
374  the example.
375 
376  particle_count - [in]
377  Number of particles in particles[] array.
378 
379  particles[] - [in]
380  List of particles this force acts on.
381 
382  Example:
383  // Hook's law spring
384  // This example if for illustration purposes.
385  // In practice it is more efficient to call ON_ParticleSystem::AddHooksSpringForce().
386 
387  struct HooksSpringForceContext
388  {
389  double m_rest_length;
390  double m_spring_constant;
391  double m_damping_constant;
392  };
393 
394  static bool HooksSpringForceFunction(
395  ON__UINT_PTR context,
396  class ON_ParticleSystem::ON_Particle** particles
397  )
398  {
399  const ON_3dVector dx = particles[1]->m_position - particles[0]->m_position;
400  const double length = dx.Length();
401  if ( length > 0.0 )
402  {
403  const ON_3dVector dv = particles[1]->m_velocity - particles[0]->m_velocity;
404  const struct HooksSpringForceContext* spring_context = (const struct HooksSpringForceContext*)context;
405  double f = spring_context->m_spring_constant*(1.0 - spring_context->m_rest_length/length)
406  + spring_context->m_damping_constant*((dv.x*dx.x + dv.y*dx.y + dv.z*dx.z)/(length*length));
407  ON_3dVector F = f*dx;
408  particles[0]->m_force += F;
409  particles[1]->m_force -= F;
410  }
411  return true;
412  }
413 
414  class ON_ParticleSystem::ON_Particle* particle1 = ...;
415  class ON_ParticleSystem::ON_Particle* particle2 = ...;
416  double rest_length = ...;
417  double spring_constant = ...;
418  double damping_constant = ...;
419  ON_ParticleSystem::ON_Particle* particles[2] = {particle1,particle2};
420  ON_Force* F = AddNaryForce(HooksSpringForceFunction,0,2,particles);
421  if ( 0 != F )
422  {
423  F->m_context.m_ptr = (ON__UINT_PTR)(&F->m_context.m_x[0]);
424  struct HooksSpringForceContext* spring_context = (struct HooksSpringForceContext* )F->m_context.m_ptr;
425  spring_context->m_rest_length = rest_length;
426  spring_context->m_spring_constant = spring_constant;
427  spring_context->m_damping_constant = damping_constant;
428  }
429 
430  Remarks:
431  To add the common damped spring force that is calculated using Hook's law, it is
432  easier to call AddHooksSpringForce().
433  */
434  class ON_ParticleSystem::ON_Force* AddNaryForce(

632  /*
633  Description:
634  Updates context.m_state.m_position and context.m_state.m_velocity
635  using Euler integration.
636 
637  Parameters:
638  context - [in / out]
639  The returned position and velocity are incremented
640  using the following calculation.
641  x1 = current position
642  v1 = current velocity
643  a1 = (current accumulated force)/(current mass)
644  // incremented position and velocity
645  x = x1 + v1*dt + 0.5*a1*dt*dt;
646  v = v1 + a1*dt;
647 
648  Remarks:
649  This is the best choice in the rare case when all forces are
650  independent of time. The canonical example when being when
651  the only force is the force of gravity and no collisions occur.
652  Note that forces whose formulae involve position or velocity are
653  almost never indepent of time. In particular, if your system
654  has drag, damping or spring forces, then this is probably
655  the worst choice for an integrator.
656  */

658  class ON_ParticleSystem::ON_IntegrateContext& context
659  );
