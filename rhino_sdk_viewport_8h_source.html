282  /*
283  Description:
284  Retrieves the exact contents of this viewport's entire z-buffer. Meaning,
285  what you see is what you get...If you want to change the output contents of the
286  z-buffer, (ie. forced shading) then you should construct your own CRhinoZBuffer
287  object using this viewport, make the appropirate CRhinoZBuffer method calls, and
288  then "Capture()" the z-buffer (see CRhinoZBuffer for details).
289  Parameters:
290  CRhinoZBuffer& : output parameter that will contain all captured z-buffer info.
291  Returns:
292  true: If z-buffer was successfully captured...
293  false: If no z-buffer was captured (or some other failure ocurred).
294 
295  Examples:
296 
297  CRhinoZBuffer zbuffer;
298 
299  // Test contents of entire zbuffer...
300  if ( pView->Viewport().GetZBuffer( zbuffer ) )
301 
302  for (int y = 0; y < zbuffer.Height(); y++)
303  for (int x = 0; x < zbuffer.Width(); x++)
304  {
305  float z = zbuffer.ZValue( x, y );
306 
307  if ( (z > 0.0) && (z < 1.0) )
308  {
309  // Then point (x,y) 'hit' something...
310  }
311  }
312 
313  // Test contents of a rectangular area within the z-buffer and convert
314  // the hit points to 3D world coordinates...
315  CRhinoZBuffer zbuffer;
316  CRect rect = some rectangular portion of the viewport...
317  ON_SimpleArray<ON_3dPoint> points;
318 
319  if ( pView->Viewport().GetZBuffer( zbuffer ) )
320  for (int y = rect.top; y < rect.bottom; y++)
321  for (int x = rect.left; x < rect.right; x++)
322  {
323  float z = zbuffer.ZValue( x, y );
324 
325  if ( (z > 0.0) && (z < 1.0) )
326  {
327  points.Append( zbuffer.WorldPoint( x, y ) );
328  }
329  }
330  */
331  bool GetZBuffer(class CRhinoZBuffer&) const;
