219 Description:
220  Create a condition that should result in a crash.
221  The intended use is for testing application crash handling.
222 Parameters:
223  crash_type - [in]
224  0: dereference nullptr data pointer
225  1: dereference bogus data pointer (0xCACAF00D)
226  2: dereference nullptr function pointer
227  3: dereference bogus function pointer (0xCACAF00D)
228  4: call abort()
229  5: call exit(99);
230  6: double divide by 0.0 - FPU exception
231  well designed apps will mask the exception but detect the error
232  7: float divide by 0.0 - FPU exception
233  well designed apps will mask the exception but detect the error
234  8: int divide by 0 - will crash
235  9: log(-1.0) - should call math error handler
236  10: double overflow - FPU exception
237  well designed apps will mask the exception but detect the error
238  11: throw(123) - well designed apps with catch the exception
239  before crashing
240 Returns:
241  Value of crash_type parameter.

249 static
250 int CrashTestHelper( int crash_type, ON_TextLog& textlog )
251 {
252  // Note: The code and calls are intentionally obtuse
253  // so that it is difficult for an optimizer to
254  // not perform the calculation.
255  int stack_int = 0;
256  int rc = crash_type;
257 
258  switch( crash_type )
259  {
260  case 0: // dereference nullptr pointer
261  rc = CrashTestHelper_DerefNullIntPtr( textlog, crash_type, &stack_int );
262  break;
263 
264  case 1: // dereference bogus pointer
265  rc = CrashTestHelper_DerefBogusIntPtr( textlog, crash_type, &stack_int );
266  break;
267 
268  case 2: // call nullptr function pointer
269  rc = CrashTestHelper_CallNullFuncPtr( textlog, crash_type, &stack_int );
270  break;
271 
272  case 3: // divide by zero
273  rc = CrashTestHelper_CallBoguslFuncPtr( textlog, crash_type, &stack_int );
274  break;
275 
276  case 4:
277  abort();
278  textlog.Print("abort() didn't crash.\n");
279  break;
280 
281  case 5:
282  exit(99);
283  textlog.Print("exit(99) didn't crash.\n");
284  break;
285 
286  case 6:
287  if ( CrashTestHelper_DivideByDoubleZero( textlog, "0.0" ) )
288  {
289  textlog.Print("Divide by double 0.0 didn't crash - exception must have been handled or ignored.\n");
290  }
291  break;
292 
293  case 7:
294  if ( CrashTestHelper_DivideByFloatZero( textlog, "0.0" ) )
295  {
296  textlog.Print("Divide by float 0.0f didn't crash - exception must have been handled or ignored.\n");
297  }
298  break;
299 
300  case 8:
301  if ( CrashTestHelper_DivideByIntZero( textlog, "0.0" ) )
302  {
303  textlog.Print("Divide by int 0 didn't crash - exception must have been handled or ignored.\n");
304  }
305  break;
306 
307  case 9:
308  if ( CrashTestHelper_LogNegativeNumber( textlog, "-1.0" ) )
309  {
310  textlog.Print("log(negative number) didn't crash - exception must have been handled or ignored.\n");
311  }
312  break;
313 
314  case 10:
315  if ( CrashTestHelper_DoubleOverflow( textlog, "1.0e200", "2.0e222" ) )
316  {
317  textlog.Print("1.0e200*2.0e222 didn't crash - exception must have been handled or ignored.\n");
318  }
319  break;
320 
321  case 11:
322  if ( CrashTestHelper_Throw( textlog, "123" ) )
323  {
324  textlog.Print("throw(123) crash - exception must have been handled or ignored.\n");
325  }
326  break;
327 
328  case 12:
329  if ( CrashTestHelper_AllocationFailure( textlog, 0xABCDABCDABCDABCDllu ) )
330  {
331  textlog.Print("Attempt to allocate 0xABCDABCDABCDABCD did not crash.\n");
332  }
333  break;
334 
335  case 13:
336  if (CrashTestHelper_StackOverflow(textlog, 0))
337  {
338  textlog.Print("Attempt to crash with stack overflow failed");
339  }
340 
341  default:
342  break;
343  }
344 
345  return rc;
346 }
