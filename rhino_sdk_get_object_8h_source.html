18 /*
19 CLASS
20 CRhinoGetObject
21 
22 Description:
23  The CRhinoGetObject class is the tool commands use to interactively
24  select objects.
25 
26 Example:
27 
28  CRhinoGetObject go;
29  go.GetObject();
30  if ( go.Result() != CRhinoGet::object )
31  ... use canceled or some other type of input was provided
32  int object_count = go.ObjectCount();
33  for ( i = 0; i < object_count; i++ )
34  {
35  CRhinoObjRef objref = go.Object(i);
36  CRhinoObject* objref.Object();
37  const ON_Geometry* geo = objref.Geometry();
38  ...
39  }
40 
41 Remarks:
42  The behavior of CRhinoGetObject::GetObject() can be highly customized
43  to accept many kinds of alternative input, to restrict what kinds of
44  object may be selected, to control how many object may be selected,
45  etc. See the detailed comments in the declarations of CRhinoGet
46  and CRhinoGetObject for more information.
47 */

74  // Description:
75  // The GEOMETRY_TYPE_FILTER enum values are bitwise or-ed
76  // together to create a filter used to specify acceptable
77  // geometry types. The geometry type filter is set
78  // by calling CRhinoGetObject::SetGeometryFilter.
79  // values = ON::object_type enum values.
80  // In an ideal world, the CRhinoGetObject::GEOMETRY_TYPE_FILTER
81  // would be removed and ON::object_type would be used everywhere.
82  enum GEOMETRY_TYPE_FILTER
83  {
84  point_object = ON::object_type::point_object, // single point
85  grip_object = ON::object_type::grip_object,
86  pointset_object = ON::object_type::pointset_object, // point cloud or point grid
87  curve_object = ON::object_type::curve_object, // curve (can be edge in a brep)
88  surface_object = ON::object_type::surface_object, // surface or brep face
89  polysrf_object = ON::object_type::polysrf_filter, // brep with 2 or more faces
90  edge_object = ON::object_type::edge_filter, // brep edge (with an associated ON_BrepTrim)
91  polyedge_object = ON::object_type::polyedge_filter, // a chain of edges and/or curves
92  loop_object = ON::object_type::loop_object, // brep loop
93  brepvertex_filter = ON::object_type::brepvertex_filter, // brep vertex, ON_SubDVertex
94  mesh_object = ON::object_type::mesh_object, // ON_Mesh
95  annotation_object = ON::object_type::annotation_object,
96  light_object = ON::object_type::light_object,
97  instance_reference = ON::object_type::instance_reference,
98  meshvertex_filter = ON::object_type::meshvertex_filter, // ON_Mesh.m_V[] index, ON_MeshTopologyVertex, ON_SubDVertex
99  meshedge_filter = ON::object_type::meshedge_filter, // ON_MeshTopologyEdge, ON_SubDEdge
100  meshface_filter = ON::object_type::meshface_filter, // selection filter for ON_Mesh.m_F[] index, ON_MeshTopologyFace, ON_SubDFace
101  meshcomponent_reference = ON::object_type::meshcomponent_reference, // ON_MeshComponentRef or ON_SubDComponentRef
102  cage_object = ON::object_type::cage_object,
103  hatch_object = ON::object_type::hatch_object,
104  morph_control_object = ON::object_type::morph_control_object,
105  subd_object = ON::object_type::subd_object, // ON_SubD, ON_SubDRef, ON_SubDComponentRef, ON_SubD....
106  phantom_object = ON::object_type::phantom_object,
107  extrusion_object = ON::object_type::extrusion_object,
108  any_object = ON::object_type::any_object
109  };

111  // Description:
112  // If an object passes the geometry TYPE filter,then the
113  // geometry ATTRIBUTE filter is applied.
114  // The geometry attributes filter is set
115  // by calling CRhinoGetObject::SetGeometryFilter.
116  enum GEOMETRY_ATTRIBUTE_FILTER
117  {
118  ////////////////////////////////////////////////////////////
119  // attributes of acceptable curve objects.
120  //
121 
122  // If you want to accept only wire or edge curves, then
123  // specify wire_curve or edge_curve, otherwise both wire
124  // and edge curves will pass the attribute filter.
125  wire_curve = 1<<0, // 3d wire curve
126  edge_curve = 1<<1, // 3d curve of a surface edge
127 
128  // If you want to accept only closed or open curves, then
129  // specify either closed_curve or open_curve. Otherwise both
130  // closed and open curves will pass the attribute filter.
131  closed_curve = 1<<2, // closed curves and edges are acceptable
132  open_curve = 1<<3, // open curves and edges are acceptable
133 
134  ////////////////////////////////////////////////////////////
135  // attributes of acceptable trimming edge objects
136  // (associated with an ON_BrepTrim).
137  //
138  // If none of these attributes are explicitly specified, then
139  // any kind of trimming edge will pass the attribute filter.
140  seam_edge = 1<<4, // seam edges are acceptable
141  manifold_edge = 1<<5, // edges with 2 difference surfaces pass
142  nonmanifold_edge = 1<<6, // edges with 3 or more surfaces pass
143  mated_edge = (1<<4)|(1<<5)|(1<<6), // any mated edge passes
144  surface_boundary_edge = 1<<7, // boundary edges on surface sides pass
145  trimming_boundary_edge = 1<<8, // boundary edges that trim a surface pass
146  boundary_edge = (1<<7)|(1<<8), // any boundary edge passes
147 
148  ////////////////////////////////////////////////////////////
149  // attributes of acceptable surface objects.
150  //
151 
152  // If you want to accept only closed or open surfaces, then
153  // specify either closed_surface or open_surface. Otherwise both
154  // closed and open surfaces will pass the attribute filter.
155  closed_surface = 1<<9,
156  open_surface = 1<<10,
157 
158  // If you want to accept only trimmed or untrimmed surfaces, then
159  // specify either trimmed_surface or untrimmed_surface. Otherwise
160  // both trimmed and untrimmed surfaces will pass the attribute
161  // filter.
162  trimmed_surface = 1<<11,
163  untrimmed_surface = 1<<12,
164 
165  // If you want to accept only sub-surfaces of (multi-surface)
166  // polysrf, then specify sub_surface. If you do not want to
167  // accept sub-surfaces, then specify top_surface. Otherwise
168  // sub-surfaces and top surfaces will pass the attribute filter.
169  sub_surface = 1<<13,
170  top_surface = 1<<14,
171 
172  ////////////////////////////////////////////////////////////
173  // attributes of acceptable polysrf objects.
174  //
175 
176  // If you want to accept only manifold or nonmanifold polysrfs,
177  // then specify manifold_polysrf or nonmanifold_polysrf. Otherwise
178  // both manifold and nonmanifold polysrfs will pass the attribute
179  // filter.
180  manifold_polysrf = 1<<15,
181  nonmanifold_polysrf = 1<<16,
182 
183  // If you want to accept only closed or open polysrfs, then
184  // specify either closed_polysrf or open_polysrf. Otherwise both
185  // closed and open polysrfs will pass the attribute filter.
186  closed_polysrf = 1<<17,
187  open_polysrf = 1<<18,
188 
189  ////////////////////////////////////////////////////////////
190  // attributes of acceptable mesh objects.
191  //
192 
193  // If you want to accept only closed or open meshs, then
194  // specify either closed_mesh or open_mesh. Otherwise both
195  // closed and open meshs will pass the attribute filter.
196  closed_mesh = 1<<19,
197  open_mesh = 1<<20,
198 
199  ////////////////////////////////////////////////////////////
200  // attributes of acceptable loop objects.
201  //
202  boundary_inner_loop = 1<<21, // all trimming edges are boundary edges
203  mated_inner_loop = 1<<22, // all trimming edges are mated
204  inner_loop = (1<<21)|(1<<22), // any inner loop is acceptable
205 
206  boundary_outer_loop = 1<<23, // all trimming edges are boundary edges
207  mated_outer_loop = 1<<24, // all trimming edges are mated
208  outer_loop = (1<<23)|(1<<24), // any outer loop is acceptable
209 
210  special_loop = (1<<25), // slit, crvonsrf, ptonsrf, etc.
211 
212  accept_all_attributes = 0xffffffff // all attributes are acceptable
213  };

245  /*
246  Description:
247  The geometry type filter controls which types of geometry
248  (points, curves, surfaces, meshes, etc.) can be selected.
249  The default geometry type filter permits selection of all
250  types of geometry.
251 
252  Returns:
253  An unsigned int "object_filter". If the bit from an
254  GEOMETRY_TYPE_FILTER enum value is set in the geometry_filter,
255  then GetObjects() will permit selection of object's
256  that have that type of geometry. If geometry_filter
257  is 0xFFFFFFFF, then GetObjects() can select all types
258  of geometry.
259 
260  Example:
261  If (GeometryFilter() | curve_object) != 0, then GetObjects()
262  will permit curves to be selected.
263 
264  See Also:
265  CRhinoGetObject::SetGeometryFilter
266  CRhinoGetObject::SetGeometryAttributeFilter
267  CRhinoGetObject::GeometryAttributeFilter
268  */

276  meshes, etc.) is provided by CRhinoGetObject::SetGeometryFilter.

279  The default attribute filter permits selection of all types of

746  }
