252  // Rtf uses UTF-16 encoding and surrogate pairs need to be properly handled.
253  // For example, the single UNICODE code point ON_UnicodeCodePoint::Wastebasket U+1F5D1 (decimal 128465)
254  // is in the RTF string as ...{\ltrch \u-10179?\u-8751?}...
255  // The UNICODE code point U+1F5D1 is encoded as a UTF-16 surrogate pair is (0xD83D, 0xDDD1).
256  // \u-10179? -> unsigned short 0xD83D
257  // \u-8751? -> unsigned short 0xDDD1
258  enum : ON__UINT16
259  {
260  m_UFT16_waiting_mark = 0xEEEE, // value must be > 0xE000 and uncommon unicode code point
261  m_UFT16_unused_mark = 0xFFFF // value must be > m_UFT16_waiting, <= 0xFFFF, and uncommon unicode code point
262  };
263  ON__INT32 m_current_UTF16_buffer_count = 0;
264  ON__UINT16 m_current_UTF16_buffer[2];
