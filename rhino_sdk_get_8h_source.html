327  //
328  // Possible results from CRhinoGetObject::GetObject(),
329  // CRhinoGetPoint::GetPoint(), etc.
330  enum result
331  {
332  no_result = 0,
333  cancel = 1, // user wants to cancel current command
334  nothing = 2, // user pressed enter - typically used to accpt defaults
335  option, // user specified an option - call Option() to get option index
336  number, // user entered a real number - call Number() to get value
337  color, // user entered a color - call Color() to get value
338  undo, // user pressed undo
339  miss, // user clicked and missed
340  point, // user picked 3d point - call Point() to get 3d point
341  point2d, // user picked 2d window point in CRhinoGetPoint::Get2dPoint()
342  // call Point2d() to get the point and View() to get the view.
343  line2d, // user picked a 2d line in CRhinoGetPoint::Get2dLine()
344  // call Line2d() to get the line and View() to get the view.
345  rect2d, // user picked a 2d rectangle in CRhinoGetPoint::Get2dRectangle()
346  // call Rectangle2d() to get the rectangle and View() to get the view.
347  object, // user selected an object - call Object() to get object
348  string, // user typed a string - call String() to get the string
349  winmsg, // Windows posted a message id that was in the list passed
350  // to CRhinoGet::AcceptWindowsMessage(). Call CRhinoGet::WndMsg()
351  // to get the message.
352  timeout, // The getter waited for the amount of time specifed
353  // in CRhinoGet::SetWaitDuration() and then gave up.
354 
355  /////////////
356  // higher level stuff
357  //
358  circle, // call CRhinoGetCircle::GetCircle() to get the circle
359  plane, // call CRhinoGetPlane::GetPlane() to get the plane
360  cylinder, // call CRhinoGetCylinder::GetCylinder() to get the cylinder
361  sphere, // call CRhinoGetSphere::GetSphere() to get the sphere
362  angle, // call CRhinoGetAngle::Angle() to get the angle in radians (CRhinoGetAngle() returns this for typed number, too)
363  distance, // call CRhinoGetDistance::Distance() to get the distance value
364  direction,// call CRhinoGetDirection::Direction() to get the direction vector
365  frame, // call CRhinoGetFrame::Frame() to get the frame that was picked
366 
367  number_preview, // internal, user is typing a number - call Number() to get value
368  point_preview, // internal, user is typing a 3d point - call Point() to get 3d point
369 
370  user1 = 0xFFFFFFFF,
371  user2 = 0xFFFFFFFE,
372  user3 = 0xFFFFFFFD,
373  user4 = 0xFFFFFFFC,
374  user5 = 0xFFFFFFFB,
375 
376  exit_rhino = 0x0FFFFFFF // stop now, do not cleaup, just return ASAP.

717  /*
718  Description:
719  Add a command line option to get numbers and automatically
720  save the value.
721  Parameters:
722  option_name - [in] english option description
723  Automatic localization uses command option string table.
724  number_value - [in/out] pointer to current value of the number.
725  If the user changes the nubmer's value, the value of
726  *number_value is changed but the current call to
727  GetPoint/GetString/... does not return.
728  option_prompt - [in] option prompt shown if the user selects
729  this option. If NULL, then option_name.m_local_option_name
730  is used as the get number prompt.
731  bIntegerValue - [in] (default=false) if true, number must be an integer
732  lower_limit - [in] (default=ON_UNSET_VALUE which means no lower limit)
733  if set, the returned number must be >= lower_limit
734  upper_limit - [in] (default=ON_UNSET_VALUE which means no upper limit)
735  if set, the returned number must be <= upper_limit
736  number_format - [in] This controls how strings are parsed to get a number
737  and how the number is displayed in the command prompt.
738
