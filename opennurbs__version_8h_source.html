129 ////////////////////////////////////////////////////////////////
130 //
131 // ON_VERSION_NUMBER_FEBDAYS(year) is a macro whose value is
132 // the number of days in the month of February in a specified
133 // year.
134 //
135 // In almost every situation, it is best to used the function
136 // call ON_DaysInMonthOfGregorianYear(year,2) to get this value.
137 // The ON_VERSION_NUMBER_FEBDAYS macro is for rare and unusual
138 // situations where the C preprocessor needs this value.
139 //
140 #define ON_VERSION_NUMBER_FEBDAYS(year) \
141  (((year) % 400) == 0 ? 29 : \
142  (((year) % 100) == 0 ? 28 : \
143  (((year) % 4) == 0 ? 29 : \
144  28)))
145 
146 ////////////////////////////////////////////////////////////////
147 //
148 // ON_VERSION_NUMBER_DAYOFYEAR(year, month, day_of_month) is a macro
149 // whose value is the cardinal day of the year for the
150 // specified year, month and day_of_month.
151 //
152 // In almost every situation, it is best to used the function call
153 // ON_DayOfGregorianYear(year,month,day_of_month) to get this value.
154 // The ON_VERSION_NUMBER_DAYOFYEAR macro is for rare and unusual
155 // situations where the C preprocessor needs this value.
156 //
157 #define ON_VERSION_NUMBER_DAYOFYEAR(year, month, day_of_month) \
158  ( (day_of_month) \
159  + ((month) >= 2 ? 31 : 0) \
160  + ((month) >= 3 ? ON_VERSION_NUMBER_FEBDAYS(year) : 0) \
161  + ((month) >= 4 ? 31 : 0) \
162  + ((month) >= 5 ? 30 : 0) \
163  + ((month) >= 6 ? 31 : 0) \
164  + ((month) >= 7 ? 30 : 0) \
165  + ((month) >= 8 ? 31 : 0) \
166  + ((month) >= 9 ? 31 : 0) \
167  + ((month) >= 10 ? 30 : 0) \
168  + ((month) >= 11 ? 31 : 0) \
169  + ((month) >= 12 ? 30 : 0) \
170  )

172 #define ON_VERSION_NUMBER_TIME(year, month, day_of_month) \
173  ((((year)-2000)*367) + (ON_VERSION_NUMBER_DAYOFYEAR(year,month,day_of_month)))

175 // branch is 0 for developer builds, odd for windows builds, and even for Mac builds
176 // When compressed into 2 bits, the type of build (Commercial/Beta/WIP) is not encoded.
177 //
178 // branch
179 // 0: developer build
180 // 1: Windows Commercial build
181 // 2: Mac Commercial build
182 // 3: Windows BETA build
183 // 4: Mac Beta build
184 // 5: Windows WIP build
185 // 6: Mac WIP build
186 //
187 // ON_VERSION_NUMBER_PLATFORM_ID(branch) =
188 // = 0: developer build
189 // 1: Windows build
190 // 2: Mac build
191 //
192 // NOTE WELL:
193 // ON_VERSION_NUMBER_PLATFORM_ID(branch) must be a value between 0 and 3.
194 #define ON_VERSION_NUMBER_PLATFORM_ID(branch) \
195  (((branch) > 0x0U) ? (0x02U - ((branch) % 0x02U)) : 0x0U)

198 //
199 // ON_VERSION_NUMBER_CTOR(major,minor,year,month,day_of_month,branch)
200 // is a macro whose value is the opennurbs version number encoding
201 // for the specified major, minor, year, month and day_of_month
202 // values.
203 //
204 // In almost every situation, it is best to used the function call
205 // ON_VersionNumberConstruct(major,minor,year,month,day_of_month,branch)
206 // to get this value. The ON_VERSION_NUMBER_CTOR macro is for
207 // rare and unusual situations where the C preprocessor needs
208 // this value.
209 //
210 #define ON_VERSION_NUMBER_CTOR(major,minor,year,month,day_of_month,branch) \
211  (0x80000000U \
212  + ((((major)*0x080U + (minor)))*0x010000U \
213  + ((ON_VERSION_NUMBER_TIME(year,month,day_of_month))))*0x04U \
214  + ((ON_VERSION_NUMBER_PLATFORM_ID(branch))))
