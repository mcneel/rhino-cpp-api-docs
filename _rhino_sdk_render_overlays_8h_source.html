28 // Rendering overlay class that creates "maskable" renderings (known as Overlays),
29 // of all of the "non-mesh based" objects in the scene. These overlays can then
30 // be added or combined with other renderings providing a general solution for
31 // those renderers that do not support some or all of the non-mesh based objects.
32 class RHINO_SDK_CLASS CRhinoRenderOverlays
33 {
34 public:
35  CRhinoRenderOverlays();
36  ~CRhinoRenderOverlays();

39  // Description:
40  // Creates all of the requested overlays in the given resolution using the
41  // given frustum and display attributes. These overlays are stored internally
42  // and can only be accessed via the "GetOverlayXXX" methods.
43  //
44  // Note: This is the starting point for using overlays. Basically you use this
45  // routine to specify ALL of the types of overlays you want to support,
46  // and then you use GetOverlayXXXX() to gather/comibine an overlay or set
47  // of overlays into one maskable entity. An overlay is made up of one or
48  // more "features", and a feature is described/defined by ERenderOverlayFeatures.
49  //
50  // Parameters:
51  // Frustum - [in]
52  // viewing frustum that specifies camera and projection from which to render
53  // the overlays. This most likely will come from one of Rhino's viewports, but
54  // doesn't necessarily have to.
55  // Attrs - [in]
56  // Display attributes that will be used when rendering the overlays. These will
57  // only define "how" things will get rendered, and not "what" gets rendered, since
58  // that is what is determined by this routine in general. For example: Isocurve thickness
59  // and color can be controlled by these attributes.
60  // nWidth - [in]
61  // The physical width in pixels that you want all overlay resolutions to be.
62  // nHeight - [in]
63  // The physical height in pixels that you want all overlay resolutions to be.
64  // nFeatures - [in]
65  // A bitwise mask (using ERenderOverlayFeatures constants) that specifies which
66  // overlays you're going/wanting to support.
67  // bCombine - [in]
68  // If "true", then all requested features will be rendered into a single ovleray.
69  // If "false", then all requested features will be rendered and maintained in
70  // their own individual overlay.
71  //
72  // Returns:
73  // TRUE if all requested features were rendered successfully into overlays.
74  // FALSE if something bad happened during the creation process.
75  //
76  // Example:
77  //
78  // unsigned int nSupportedFeatures = ROF_CURVES|ROF_ISOS|ROF_EDGES|ROF_POINTS;
79  //
80  // CRhinoRenderOverlays PostRender;
81  //
82  // if ( PostRender.CreateOverlays( view, attrs, 1024, 768, nSupportedFeatures, false ) )
83  // {
84  // unsigned int nUsedFeatures = 0;
85  //
86  // if ( bShowCurves )
87  // nUsedFeatures |= ROF_CURVES;
88  // if ( bShowIsos )
89  // nUsedFeatures |= ROF_ISOS;
90  // if ( bShowEdges )
91  // nUsedFeatures |= ROF_EDGES;
92  // if ( bShowPoints )
93  // nUsedFeatures |= ROF_POINTS;
94  //
95  // CRhinoDib overlay;
96  //
97  // if ( PostRender.GetOverlay( overlay, nUsedFeatures ) )
98  // {
99  // PostRender.CombineOverlay( overlay, SOME_FINAL_RENDERING_IMAGE );
100  // }
101  // }
102  //
103  bool CreateOverlays(const ON_3dmView& Frustum,
104  CDisplayPipelineAttributes Attrs,
105  int nWidth,
106  int nHeight,
107  unsigned int nFeatures,
108  bool bCombine=true);

111  // Description:
112  // Creates the requested overlay based on the specified features.
113  //
114  // Parameters:
115  // Overlay - [out]
116  // The created overlay containing all of the requested features. This will be
117  // a 32bit DIB containing all requested features with alpha channel. The alpha channel
118  // will include anti-aliasing values used when the overlays were initially created.
119  //
120  // nFeatures - [in]
121  // The bitwise mask made up of ERenderOverlayFeatures constants that defines
122  // which features should be rendered into the Overlay. (Note: This mask
123  // must be equal to or a subset of the mask used when calling "CreateOverlays").
124  //
125  // Returns:
126  // TRUE if the overlay with the requested features was successfully created.
127  // FALSE if one or more the requested features could not be successfully rendered
128  // into the overlay. This can happen if you specify a feature that was not also
129  // specified in the call to "CreateOverlays" prior to calling this method.
130  //
131  // Note: A successful call to "CreateOverlays" must be made prior to calling this method.
132  //
133  bool GetOverlay(CRhinoDib& Overlay, unsigned int nFeatures) const;
134

136  // Description:
137  // Creates the requested overlay based on the specifed features in a grayscale format.
138  //
139  // Parameters:
140  // Overlay - [out]
141  // The created overlay containing all of the requested features. This will be
142  // a 24bit RGB DIB containing all requested features, but will be in a grayscale
143  // format (i.e. R = G = B ) regardless of what attribute settings were used
144  // when creating the overlays initially.
145  //
146  // nFeatures - [in]
147  // The bitwise mask made up of ERenderOverlayFeatures constants that defines
148  // which features should be rendered into the Overlay. (Note: This mask
149  // must be equal to or a subset of the mask used when calling "CreateOverlays").
150  //
151  // Returns:
152  // TRUE if the overlay with the requested features was successfully created.
153  // FALSE if one or more the requested features could not be successfully rendered
154  // into the overlay. This can happen if you specify a feature that was not also
155  // specified in the call to "CreateOverlays" prior to calling this method.
156  //
157  // Note: A successful call to "CreateOverlays" must be made prior to calling this method.
158  //
159  bool GetOverlayAsAlpha(CRhinoDib& Overlay, unsigned int nFeatures) const;

161  // Description:
162  // Method used to determine whether or not certain overlays can be created using the
163  // specified features.
164  //
165  // Parameters:
166  // nFeatures - [in]
167  // The bitwise mask made up of ERenderOverlayFeatures constants that defines
168  // which features should be considered when performaing the query. (Note: This mask
169  // must be equal to or a subset of the mask used when calling "CreateOverlays").
170  //
171  // Returns:
172  // TRUE if the specified features would result in a successful overlay creation.
173  // FALSE if the specified features would not create a valid overlay.
174  //
175  // Note: A successful call to "CreateOverlays" must be made prior to calling this method.
176  //
177  bool HasOverlay(unsigned int nFeatures) const;

179  // Description:
180  // Combines two like overlays into one.
181  //
182  // Parameters:
183  // Source - [in]
184  // Contains the primary source information of one of the overlays to be combined.
185  // Destination - [in out]
186  // Contains the secondary source information of the other overlay to be combined,
187  // but also will have its contents modified/updated to resulting combination.
188  //
189  // Returns:
190  // TRUE if the combination of the two overlays was successful, and the results are
191  // placed in Destination.
192  // FALSE if the combination failed. This might be due to a mismatch in resolution or
193  // color depth between the two overlays.
194  //
195  // Note: Both Source and Destination resolutions and color depths MUST be exactly the same.
196  //
197  bool CombineOverlays(const CRhinoDib& Source, CRhinoDib& Destination);
