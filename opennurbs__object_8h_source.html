297  ON_OBJECT_DECLARE( <classname> );

300 have exactly one of the following implementation macros in
301 a .cpp file.
302  Classes with a pure virtual function:
303  ON_VIRTUAL_OBJECT_IMPLEMENT( <classname>, <basclassname>, <classuuid> )
304  Classes with an operator= and copy constructor.
305  ON_OBJECT_IMPLEMENT( <classname>, <basclassname>, <classuuid> )
306  Classes with an operator=, but no copy constructor.
307  ON_OBJECT_IMPLEMENT_NO_COPYCTOR( <classname>, <basclassname>, <classuuid> )
308 */

311  static void* m_s_##cls##_ptr; \
312  \

314  /* OpenNURBS class run-time type information */ \
315  static const ON_ClassId m_##cls##_class_rtti; \
316  \

318  static cls * Cast( ON_Object* ); \
319  \
320  /*OpenNURBS platfrom independent dynamic cast*/ \
321  static const cls * Cast( const ON_Object* ); \
322  \
323  /*Returns: OpenNURBS run-time type information.*/ \
324  ON_OBJECT_DECLARE_VIRTUAL const ON_ClassId* ClassId() const ON_OBJECT_DECLARE_OVERRIDE; \

349 /*
350 Classes derived from ON_Object that are pure virtual classes,
351 or do not have a valid default constructor, valid operator new
352 or valid operator= must use ON_VIRTUAL_OBJECT_IMPLEMENT in their
353 implementation. Classes implemented with ON_VIRTUAL_OBJECT_IMPLEMENT
354 cannot be serialized using ON_BinaryArchive::ReadObject()/WriteObject()
355 or duplicated using ON_Object::Duplicate().
356 The Cast() and ClassId() members work on classes implemented with
357 ON_VIRTUAL_OBJECT_IMPLEMENT, ON_OBJECT_IMPLEMENT or
358 ON_OBJECT_IMPLEMENT_NO_COPYCTOR
359 */
360 #define ON_VIRTUAL_OBJECT_IMPLEMENT( cls, basecls, uuid ) \
361  void* cls::m_s_##cls##_ptr = nullptr; \
362  const ON_ClassId cls::m_##cls##_class_rtti(#cls,#basecls,0,uuid);\
363  cls * cls::Cast( ON_Object* p) {return(p&&p->IsKindOf(&cls::m_##cls##_class_rtti))?static_cast< cls *>(p):nullptr;} \
364  const cls * cls::Cast( const ON_Object* p) {return(p&&p->IsKindOf(&cls::m_##cls##_class_rtti))?static_cast<const cls *>(p):nullptr;} \
365  const ON_ClassId* cls::ClassId() const {return &cls::m_##cls##_class_rtti;} \
366  bool cls::CopyFrom(const ON_Object*) {return false;} \
367  cls * cls::Duplicate() const {return static_cast< cls *>(this->Internal_DeepCopy());} \
368  ON_Object* cls::Internal_DeepCopy() const {return nullptr;}
369 
370 /*
371 Classes derived from ON_Object that have a valid default constructor,
372 valid copy constructor, operator new and operator= can use
373 ON_OBJECT_IMPLEMENT in their implementation. Classes implemented
374 with ON_OBJECT_IMPLEMENT can be created from their run-time type
375 information id and their Duplicate() function will use the class's
376 copy constructor to create a deep copy.
377 */
378 #define ON_OBJECT_IMPLEMENT( cls, basecls, uuid ) \
379  void* cls::m_s_##cls##_ptr = nullptr; \
380  static ON_Object* CreateNew##cls() {return new cls();} \
381  const ON_ClassId cls::m_##cls##_class_rtti(#cls,#basecls,CreateNew##cls,uuid);\
382  cls * cls::Cast( ON_Object* p) {return(p&&p->IsKindOf(&cls::m_##cls##_class_rtti))?static_cast< cls *>(p):nullptr;} \
383  const cls * cls::Cast( const ON_Object* p) {return(p&&p->IsKindOf(&cls::m_##cls##_class_rtti))?static_cast<const cls *>(p):nullptr;} \
384  const ON_ClassId* cls::ClassId() const {return &cls::m_##cls##_class_rtti;} \
385  bool cls::CopyFrom( const ON_Object* src){const cls * s=cls::Cast(src); if ( nullptr != this && nullptr != s) {*this = *s; return true;}return false;} \
386  cls * cls::Duplicate() const {return static_cast< cls *>(this->Internal_DeepCopy());} \
387  ON_Object* cls::Internal_DeepCopy() const {return new cls (*this);}
