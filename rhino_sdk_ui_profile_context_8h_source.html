22 Description:
23  A collection of key value string pairs, each key name must be unique, key
24  names are case sensitive.
25 */
26 class RHINO_SDK_CLASS CRhinoStringDictionary
27 {
28 public:
29  /*
30  Description:
31  Constructor
32  */
33  CRhinoStringDictionary();
34  /*
35  Description:
36  Call this method to get the number of items in the dictionary.
37  Returns:
38  Returns the number of items in the dictionary
39  */
40  int Count() const;
41  /*
42  Description:
43  Call this method to clear the dictionary and set its count to 0.
44  */
45  void Empty();
46  /*
47  Description:
48  Call this method to add or replace a key in the dictionary. The key name
49  is case sensitive. If there is currently a matching key then the value
50  for that key is replaced with the new value.
51  Parameters:
52  key - [in] Case sensitive key name
53  value - [in] Value associated with the specified key
54  Returns:
55  Returns the index of the item added or modified if successful or -1 on
56  error.
57  */
58  int Add(const wchar_t* key, const wchar_t* value);
59  /*
60  Description:
61  Call this method to delete a specific key, the key name is case sensitive.
62  Parameters:
63  key - [in] Case sensitive key name
64  Returns:
65  Returns true if the key was found in the dictionary and successfully deleted
66  or false on error.
67  */
68  bool Delete(const wchar_t* key);
69  /*
70  Description:
71  Call this method to delete the key value pair at the specified index.
72  Parameters:
73  index - [in] Zero based index
74  Returns:
75  Returns true if the index was valid and the key value pair was successfully
76  deleted otherwise; returns false on error.
77  */
78  bool Delete(int index);
79  /*
80  Description:
81  Call this method to modify the value associated with the specified key,
82  the key is case sensitive.
83  Parameters:
84  key - [in] Case sensitive key name
85  value - [in] New value
86  Returns:
87  Returns true if the key was found and the value was successfully set.
88  */
89  bool Set(const wchar_t* key, const wchar_t* value);
90  /*
91  Description:
92  Call this method to get the 0 based index of the specified key.
93  Parameters:
94  key - [in] Case sensitive key name
95  Returns:
96  Returns a value greater than 0 and less than Count() if successful
97  otherwise; returns -1.
98  */
99  int IndexOf(const wchar_t* key) const;
100  /*
101  Description:
102  Array operator that allows access to key values
103  Parameters:
104  key - [in] Case sensitive key name
105  Returns:
106  Returns the value if found or NULL on error.
107  */
108  const wchar_t* operator[](const wchar_t* key);
109  /*
110  Description:
111  Call this method to get a key name at a specific index.
112  Parameters:
113  index - [in] Zero based index
114  Returns:
115  Returns the key name string if the index is greater than 0 and less than Count().
116  */
117  const wchar_t* Key(int index) const;
118  /*
119  Description:
120  Call this method to get a value at a specific index.
121  Parameters:
122  index - [in] Zero based index
123  Returns:
124  Returns the value string if the index is greater than 0 and less than Count().
125  */
126  const wchar_t* Value(int index) const;
127  /*
128  Description:
129  Call this method to get the value for the specified key name, the key name
130  string is case sensitive.
131  Parameters:
132  key - [in] Case sensitive key name
133  Returns:
134  Returns the value string if the key name is found in the dictionary otherwise;
135  returns NULL.
136  */
137  const wchar_t* Value(const wchar_t* key) const;
138  /*
139  Description:
140  Call this method to gain direct access to the list of keys in the dictionary.
141  Returns:
142  Returns a const reference to the key list.
143  */
144  const ON_ClassArray<ON_wString>& Keys() const;
145  /*
146  Description:
147  Call this method to gain direct access to the list of values in the dictionary.
148  Returns:
149  Returns a const reference to the value list.
150  */
151  const ON_ClassArray<ON_wString>& Values() const;
152 
153 private:
154  friend class CRhSettingsHooks;
155 
156  ON_ClassArray<ON_wString>m_keys;
157  ON_ClassArray<ON_wString>m_values;
158 };
