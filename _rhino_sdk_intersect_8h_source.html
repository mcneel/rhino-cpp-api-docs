23  /*
24  There are two valid curve-curve intersection events
25  CCX_POINT - curves intersect in an isolated point
26  a[0] = a[1] = first curve parameter
27  A[0] = A[1] = intersection point on first curve
28  b[0] = b[1] = second curve parmeter
29  B[0] = B[1] = intersection point on second curve
30 
31  CCX_OVERLAP
32  (a[0],a[1]) = first curve parameter range
33  A[0] = intersection start point on first curve
34  A[1] = intersection end point on first curve
35  (b[0],b[1]) = second curve parameter range
36  B[0] = intersection start point on second curve
37  B[1] = intersection end point on second curve
38  */
39  enum EVENT_TYPE

41  CCX_NO_EVENT = 0,
42  CCX_POINT = 1, // curve-curve intersection point
43  CCX_OVERLAP = 2, // curve-curve intersection overlap
44  CCX_EVENT_TYPE__FORCE__INT = 0xFFFFFFFF
45  };
