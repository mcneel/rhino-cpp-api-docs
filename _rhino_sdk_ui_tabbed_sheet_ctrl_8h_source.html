314 class RHINO_SDK_CLASS CRhinoUiTabbedSheetCtrlMsg
315 {
316 public:
317  enum message_id
318  {
319  mid_sel_changing = 1, // Message sent when the current tab is about to change
320  mid_sel_changed, // Message sent when the current tab has been changed
321  mid_delete_item, // Message sent just prior to deleting an item
322  mid_begin_label_edit, // Message sent prior to editing an item label, return 1 to allow
323  // Editing to continue or 0 to disable editing for this item
324  mid_end_label_edit, // Message sent when in-place edit control is hidden and text has
325  // changed. Return 1 to commit change or 0 to abort.
326  mid_right_mouse_down_on_tab, // The right mouse button has gone down over a tab. The tab
327  // will be made current prior to this message being sent.
328  mid_begin_tab_drag, // Message sent just prior to starting to drag a tab. If the m_bool property
329  // is not set to true then the drag action is aborted. By default m_bool is
330  // true.
331  mid_tab_dropped, // Message sent when a tab is dragged and dropped someplace other
332  // than on another tab. m_pt Will contain screen point for drop action.
333  mid_tab_drag_moved, // Message sent when a tab is dragged to a new position in the control.
334  // m_iItem Will be equal to new position, m_nPrevSel will be equal to
335  // previous position.
336  left_mouse_down_on_add_button, // The left mouse went down on the add button, the "+" after the last tab
337  add_button_click, // The left mouse went down and up on the add button, the "+" after the last tab
338  // for internal use only, this allows adding additional values without
339  // breaking the SDK in the future
340  force_32bit_message_id = 0xFFFFFFFF
341  };
