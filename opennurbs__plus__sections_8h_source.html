66  // The "base plane" equation is stored in m_plane_equation.
67  //
68  // If m_offset_spacing != 0, then sections are
69  // calculated at the planes
70  // a*x + b*y + c*z + (d - n*m_plane_delta),
71  // where n is an integer value. (The value n*m_plane_delta
72  // is subtracted so that positive values of n*m_plane_delta
73  // move the plane in the direction of the plane's normal
74  // vector and programmers who are not familier with translating
75  // plane equations are more likely to get what they expect.)
76  //
77  // If m_plane_delta_limits[0] < m_plane_delta_limits[1], then
78  // n is restricted to the range,
79  // m_plane_delta_limits[0] <= n < m_plane_delta_limits[1].
80  // Otherwise all values of n are used and m_max_plane_count
81  // is used to limit the maximum number of section planes.
82  //
83  // Observe that m_plane_delta is a 3d distance between
84  // the planes only when (x,y,z) is a unit vector.
85  //
86  // If m_plane_delta_limits[0] <= m_plane_delta_limits[1]
87  // and m_plane_delta > 0, then m_max_plane_count is used
88  // to limit the number of section planes.
89  // If m_max_plane_count = 0, then at most 100 section planes
90  // will be calculated for any one object.
91  // If m_max_plane_count > 0, then the value of m_max_plane_count
92  // will be used to limit the number of section planes for
93  // any one object.
94  //
95  // Use ON_SectionAnalysisPlane::GetPlaneIndices() to easily
96  // get a finite list of section planes that intersect a
97  // bounding box.
98  ON_PlaneEquation m_plane_equation;
99  double m_plane_delta;
100  int m_plane_delta_limits[2];
101  int m_max_plane_count;
102 private:
103  int m_reserved1;
104 public:

109  set of sectional analysis planes that will intersect
110  the input bounding box.
111 
112  Example:
113  int plane_index0 = 0;
114  int plane_index1 = 0;
115  GetPlaneIndices(bbox,tolerance,&plane_index0,&plane_index1);
116  for ( int i = plane_index0; i < plane_index1; i++ )
117  {
118  ON_PlaneEquation pe(m_plane_equation);
119  pe.d += i*m_plane_delta;
120  }
121  */
122  bool GetPlaneIndices(
123  ON_BoundingBox bbox,
124  double tolerance,
125  int* plane_index0,
126  int* plane_index1
127  ) const;
