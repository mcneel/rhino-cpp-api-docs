198  // On little endian (Intel) computers, m_color has the same byte order

200  // On little endian computers, m_color = 0xaabbggrr as an unsigned int value.

206  unsigned int m_color = 0;

208  // m_colorComponent is a 4 unsigned byte array in RGBA order
