18 /*
19 Description:
20  CRhinoGetPoint is the Rhino SDK user interface tool used to get points.
21  It is derived from CRhinoGet which is used to provide the command prompt
22  and option setting tools that are common to all the Rhino SDK getting
23  tools.
24 
25  Step 1:
26  Use the CRhinoGet member functions to specify command prompt,
27  command options, and what types of non-point input are acceptable.
28  For example, if you want the user to be able to either pick a point
29  or type a number, then you would call CRhinoGet::AcceptNumber(true).
30  The default command prompt is a localized "Pick point" prompt.
31 
32  Step 2:
33  Use the CRhinoGetPoint member functions to customize point
34  picking behavior.
35 
36  Step 3:
37  Call CRhinoGetPoint::GetPoint();
38 
39  Step 4:
40  Examine the value returned by CRhinoGet::Result() and react
41  accordingly.
42 
43 Example:
44  CRhinoCommand::result my_center_radius_getter(
45  ON_3dPoint& my_center,
46  double* my_radius
47  )
48  {
49  double radius = 1.0;
50  ON_3dPoint center(0,0,0);
51 
52  CRhinoGetPoint gp;
53  gp.SetCommandPrompt( L"Center point" );
54  gp.SetCommandPromptDefault( L"(0,0,0)" );
55  gp.AddCommandOptionNumber( L"radius", &radius, "New radius",
56  false, ON_ZERO_TOLERANCE);
57  gp.AcceptNothing(true);
58  for (;;)
59  {
60  gp.GetPoint();
61  switch (gp.Result())
62  {
63  case CRhinoGet::cancel:
64  return CRhinoCommand::cancel; // user pressed ESCAPE
65  break;
66 
67  case CRhinoGet::nothing:
68  // user pressed enter - use default center value
69  break;
70 
71  case CRhinoGet::option:
72  // user changed radius value - radius automatically updated
73  continue;
74  break;
75 
76  case CRhinoGet::point:
77  center = gp.Point();
78  break;
79 
80  default:
81  continue;
82  break;
83  }
84  break;
85  }
86  my_center = center;
87  *my_radius = radius;
88  return CRhinoCommand::success;
89  }
90 
91 See Also:
92  CRhinoGetObject, CRhinoGetString, CRhinoGetInteger, CRhinoGetNumber, CRhinoGetOption
93 */
